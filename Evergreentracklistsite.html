const modalBack = document.getElementById('modalBack');

function openModal(track){

  document.getElementById('modalTitle').innerText = track.title;

/* ------------------ Mode & meter ------------------ */

const releaseDate = new Date('2025-12-01T00:00:00');

const teaserStart = new Date('2025-06-01T00:00:00');

let mode = (new Date() >= releaseDate) ? 'released' : 'teaser';

const stem = document.getElementById('stem');

const modeLabel = document.getElementById('modeLabel');

const releaseNote = document.getElementById('releaseNote');

const toggle = document.getElementById('toggle');

const listenBtn = document.getElementById('listenBtn');

const unlockNote = document.getElementById('unlockNote');

function updateMeter(){

  const now = new Date();

  const total = releaseDate - teaserStart;

  const elapsed = Math.max(0, Math.min(total, now - teaserStart));

  const pct = Math.round((elapsed/total) * 100);

  const heightPct = Math.max(8, Math.min(100, pct));

  if(stem) stem.style.height = heightPct + '%';

  modeLabel.innerText = (mode === 'teaser') ? 'Teaser Mode' : 'Released';

  releaseNote.innerText = (mode === 'teaser') ? `Full story unlocks on ${releaseDate.toDateString()}` : 'Full Evergreen story now live';

}

function updateModalByMode(){

  if(mode === 'released'){

    if(listenBtn) listenBtn.style.display = 'inline-block';

    if(unlockNote) unlockNote.innerText = 'Full story & lyrics available.';

  } else {

    if(listenBtn) listenBtn.style.display = 'none';

    if(unlockNote) unlockNote.innerText = 'This is a chapter-style description — the full story and lyrics unlock on the release date.';

  }

}

toggle.addEventListener('click', ()=>{

  mode = (mode === 'teaser') ? 'released' : 'teaser';

  toggle.innerText = (mode === 'teaser') ? 'Switch to Released' : 'Switch to Teaser';

  updateMeter();

});

/* ------------------ Countdown (Nov 28) with sprout animation every second ------------------ */

const countdownEl = document.getElementById('countdown');

// target: November 28 (this year)

const nowYear = new Date().getFullYear();

const targetStamp = new Date(`${nowYear}-11-28T00:00:00`).getTime();

function updateCountdown(){

  const now = Date.now();

  let diff = targetStamp - now;

  if(diff <= 0){

    countdownEl.innerHTML = `🌲 It's Evergreen Day 🌲`;

    return;

  }

  const days = Math.floor(diff / (1000*60*60*24));

  const hours = Math.floor((diff % (1000*60*60*24)) / (1000*60*60));

  const minutes = Math.floor((diff % (1000*60*60)) / (1000*60));

  const seconds = Math.floor((diff % (1000*60)) / 1000);

  // sprout animation element for seconds

  countdownEl.innerHTML = `${days}d ${hours}h ${minutes}m <span class="evergreen-second"> ${seconds}s</span>`;

}

updateCountdown();
